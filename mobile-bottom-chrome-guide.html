// 一级页面的三个主导航。真实项目里可以换成自己的首页、收藏、我的。
enum class RootTab {
    Home,
    Favorites,
    Me,
}

// 二级页面类型。这里故意只保留四种，用来演示两类二级页面：
// Search/AlbumDetail 是“普通二级页”，Player/Settings 是“沉浸二级页”。
sealed interface SecondaryScreen {
    data object Search : SecondaryScreen
    data object AlbumDetail : SecondaryScreen
    data object Player : SecondaryScreen
    data object Settings : SecondaryScreen
}

// Mini Player 需要显示的最小歌曲信息。
// 真实音乐 App 可以继续增加封面、时长、专辑名等字段。
data class Song(
    val id: String,
    val title: String,
    val artist: String,
)

// BottomChromePlacement 只管一件事：底部整体容器应该停在哪里。
// 注意：这里控制的是“父容器”位置，不是单独控制 MiniPlayer 或 BottomNavigation。
enum class BottomChromePlacement {
    // 一级页：MiniPlayer 和 BottomNavigation 都在屏幕内。
    TopLevel,
    // 普通二级页：父容器整体下移一个 Tab 高度，只露出 MiniPlayer。
    MiniPlayerOnly,
    // 沉浸页：父容器整体下移 MiniPlayer + Tab 的总高度，完全藏起来。
    Hidden,
}

// ContentBottomSpace 只管页面内容底部应该留多少空间。
// 这样可以避免底部控件挡住列表，也避免控件隐藏后底部还空一大块。
enum class ContentBottomSpace {
    TopLevel,
    SecondaryWithMiniPlayer,
    Fullscreen,
}

// AppChromeMode 是“当前页面的底部规则包”。
// 一个页面不是只决定 Tab 显不显示，还要同时决定底部容器位置和内容留白。
enum class AppChromeMode(
    // BottomNavigation 是否允许显示和点击。
    val showsBottomNavigation: Boolean,
    // BottomChrome 父容器的位置。
    val bottomChromePlacement: BottomChromePlacement,
    // 页面内容底部留白策略。
    val contentBottomSpace: ContentBottomSpace,
) {
    // 一级页：Tab 显示，父容器在底部，内容留出完整底部区域。
    TopLevel(
        showsBottomNavigation = true,
        bottomChromePlacement = BottomChromePlacement.TopLevel,
        contentBottomSpace = ContentBottomSpace.TopLevel,
    ),
    // 普通二级页：Tab 不显示，只露出 MiniPlayer。
    SecondaryWithMiniPlayer(
        showsBottomNavigation = false,
        bottomChromePlacement = BottomChromePlacement.MiniPlayerOnly,
        contentBottomSpace = ContentBottomSpace.SecondaryWithMiniPlayer,
    ),
    // 沉浸二级页：MiniPlayer 和 Tab 都不显示。
    SecondaryFullscreen(
        showsBottomNavigation = false,
        bottomChromePlacement = BottomChromePlacement.Hidden,
        contentBottomSpace = ContentBottomSpace.Fullscreen,
    ),
}

// NavigationState 是当前页面位置。
// rootTab 记录当前一级 Tab；secondaryScreen 为 null 表示当前就在一级页。
data class NavigationState(
    val rootTab: RootTab = RootTab.Home,
    val secondaryScreen: SecondaryScreen? = null,
) {
    // 这是整套方案最重要的一行思想：
    // 页面只告诉我们自己是谁，chromeMode 自动推导底部区域应该怎么表现。
    val chromeMode: AppChromeMode = when (secondaryScreen) {
        // 没有二级页面，就是首页/收藏/我的这样的一级页。
        null -> AppChromeMode.TopLevel
        // 搜索和专辑详情属于普通二级页：隐藏 Tab，只露出 MiniPlayer。
        SecondaryScreen.Search,
        SecondaryScreen.AlbumDetail,
        -> AppChromeMode.SecondaryWithMiniPlayer
        // 播放页和设置页属于沉浸页：整个 BottomChrome 都隐藏。
        SecondaryScreen.Player,
        SecondaryScreen.Settings,
        -> AppChromeMode.SecondaryFullscreen
    }
}

// 所有和底部高度有关的数字集中放在这里。
// 新人不要把 68.dp、78.dp 散落到很多地方，否则后面改高度会很痛苦。
object BottomChromeDimens {
    val MiniPlayerHeight: Dp = 68.dp
    val BottomNavHeight: Dp = 78.dp
}

// 这是这份独立教程的入口组件。
// 在 App() 里调用 BottomChromeStandaloneLesson() 就可以看到完整演示。
@Composable
fun BottomChromeStandaloneLesson() {
    // 当前选中的一级 Tab。点击 BottomNavigation 时会改变它。
    var rootTab: RootTab by remember { mutableStateOf(RootTab.Home) }
    // 当前是否进入二级页面。null 表示停留在一级页。
    var secondaryScreen: SecondaryScreen? by remember { mutableStateOf(null) }
    // 当前是否正在播放。这里只用于演示“播放/暂停”文字变化。
    var isPlaying: Boolean by remember { mutableStateOf(true) }

    // 把当前页面位置组合成 NavigationState。
    val navigationState = NavigationState(
        rootTab = rootTab,
        secondaryScreen = secondaryScreen,
    )

    // 从导航状态推导出底部 Chrome 的显示策略。
    // 后面页面内容、BottomChrome 都应该读这个 chromeMode，避免各写各的判断。
    val chromeMode: AppChromeMode = navigationState.chromeMode

    // 演示用的当前歌曲。真实项目里通常来自播放器状态。
    val currentSong = Song(
        id = "demo",
        title = "海边的梦",
        artist = "旅行团乐队",
    )
    MaterialTheme {
        // 根 Box 负责把页面内容和 BottomChrome 叠在一起。
        // 页面内容先放进去，BottomChrome 再 align 到底部。
        Box(
            modifier = Modifier
                .fillMaxSize()
                .background(Color(0xFFF3F7F8)),
        ) {
            // 中间页面内容：这里提供三个按钮，方便你切换三种底部状态。
            DemoPageContent(
                secondaryScreen = secondaryScreen,
                onTopLevel = { secondaryScreen = null },
                onMiniOnly = { secondaryScreen = SecondaryScreen.AlbumDetail },
                onFullscreen = { secondaryScreen = SecondaryScreen.Player },
            )

            // 真正的底部整体容器。
            // 注意：这里只调用 BottomChrome，不再分别调用 MiniPlayer 和 BottomNavigation。
            BottomChrome(
                song = currentSong,
                isPlaying = isPlaying,
                placement = chromeMode.bottomChromePlacement,
                showsBottomNavigation = chromeMode.showsBottomNavigation,
                rootTab = rootTab,
                // 点击 MiniPlayer 时进入播放页，也就是沉浸页。
                onOpen = { secondaryScreen = SecondaryScreen.Player },
                onToggle = { isPlaying = !isPlaying },
                onPrev = { },
                onQueue = { },
                // 点击任意 Tab 时回到一级页，所以 secondaryScreen 要清空。
                onRootTab = { tab ->
                    rootTab = tab
                    secondaryScreen = null
                },
                modifier = Modifier.align(Alignment.BottomCenter),
            )
        }
    }
}

// BottomChrome 是本教程的核心组件。
// 它是“底部托盘”：里面装着 MiniPlayer 和 BottomNavigation，并且只有它负责上下移动。
@Composable
private fun BottomChrome(
    song: Song,
    isPlaying: Boolean,
    placement: BottomChromePlacement,
    showsBottomNavigation: Boolean,
    rootTab: RootTab,
    onOpen: () -> Unit,
    onToggle: () -> Unit,
    onPrev: () -> Unit,
    onQueue: () -> Unit,
    onRootTab: (RootTab) -> Unit,
    modifier: Modifier = Modifier,
) {
    // stackOffset 是整个底部托盘的 Y 方向偏移。
    // 重点：只动画这个父容器，不要让 MiniPlayer 和 BottomNavigation 各自动。
    val stackOffset: Dp by animateDpAsState(
        targetValue = when (placement) {
            // 一级页：托盘不偏移，MiniPlayer + Tab 都露出来。
            BottomChromePlacement.TopLevel -> 0.dp
            // 普通二级页：托盘下移一个 Tab 高度，所以 Tab 掉到屏幕外，只剩 MiniPlayer。
            BottomChromePlacement.MiniPlayerOnly -> BottomChromeDimens.BottomNavHeight
            // 沉浸页：托盘下移 MiniPlayer + Tab 的总高度，所以两者都离开屏幕。
            BottomChromePlacement.Hidden -> BottomChromeDimens.BottomNavHeight + BottomChromeDimens.MiniPlayerHeight
        },
        animationSpec = tween(durationMillis = 780, easing = FastOutSlowInEasing),
        label = "BottomChromeOffset",
    )
    Box(
        modifier = modifier
            .fillMaxWidth()
            .navigationBarsPadding(),
    ) {
        // 这个内部 Box 就是托盘本体，高度等于 MiniPlayer + BottomNavigation。
        // offset(y = stackOffset) 让整个托盘一起移动。
        Box(
            modifier = Modifier
                .align(Alignment.BottomCenter)
                .fillMaxWidth()
                .height(BottomChromeDimens.MiniPlayerHeight + BottomChromeDimens.BottomNavHeight)
                .offset(y = stackOffset),
        ) {
            // MiniPlayer 固定贴在托盘顶部。
            MiniPlayer(
                song = song,
                isPlaying = isPlaying,
                onOpen = onOpen,
                onToggle = onToggle,
                onPrev = onPrev,
                onQueue = onQueue,
                modifier = Modifier.align(Alignment.TopCenter),
            )
            // BottomNavigation 固定贴在托盘底部。
            BottomNavigation(
                rootTab = rootTab,
                isEnabled = showsBottomNavigation,
                onRootTab = onRootTab,
                modifier = Modifier.align(Alignment.BottomCenter),
            )
        }
    }
}

// MiniPlayer 只负责画播放器小条。
// 它不应该知道自己在一级页还是二级页，也不应该写上下移动动画。
@Composable
private fun MiniPlayer(
    song: Song,
    isPlaying: Boolean,
    onOpen: () -> Unit,
    onToggle: () -> Unit,
    onPrev: () -> Unit,
    onQueue: () -> Unit,
    modifier: Modifier = Modifier,
) {
    Surface(
        modifier = modifier
            .fillMaxWidth()
            .padding(horizontal = 22.dp)
            .height(BottomChromeDimens.MiniPlayerHeight)
            .clickable(onClick = onOpen),
        shape = RoundedCornerShape(18.dp),
        color = Color.White,
        shadowElevation = 12.dp,
    ) {
        // 横向排列：封面色块、歌曲文字、三个操作文字。
        Row(
            modifier = Modifier
                .fillMaxSize()
                .padding(horizontal = 12.dp),
            horizontalArrangement = Arrangement.spacedBy(12.dp),
            verticalAlignment = Alignment.CenterVertically,
        ) {
            // 这里用绿色方块模拟专辑封面，真实项目可以换成 Image。
            Box(
                modifier = Modifier
                    .size(46.dp)
                    .clip(RoundedCornerShape(9.dp))
                    .background(Color(0xFF17B59E)),
            )
            // 歌曲标题和歌手名，占据中间剩余空间。
            Column(modifier = Modifier.weight(1f)) {
                Text(text = song.title, fontWeight = FontWeight.Bold)
                Text(text = song.artist, color = Color(0xFF7D858E))
            }
            // 这些文字按钮只是为了让独立版能跑通；真实项目可以换成图标。
            Text(text = "上一", modifier = Modifier.clickable(onClick = onPrev))
            Text(text = if (isPlaying) "暂停" else "播放", modifier = Modifier.clickable(onClick = onToggle))
            Text(text = "队列", modifier = Modifier.clickable(onClick = onQueue))
        }
    }
}

// BottomNavigation 只负责画三个一级 Tab。
// 它不写上下移动动画，只用 isEnabled 控制隐藏状态下不能误触。
@Composable
private fun BottomNavigation(
    rootTab: RootTab,
    isEnabled: Boolean,
    onRootTab: (RootTab) -> Unit,
    modifier: Modifier = Modifier,
) {
    Surface(
        modifier = modifier
            .fillMaxWidth()
            .height(BottomChromeDimens.BottomNavHeight),
        color = Color.White,
        shadowElevation = 0.dp,
    ) {
        Row(
            modifier = Modifier
                .fillMaxSize()
                .padding(horizontal = 24.dp),
            horizontalArrangement = Arrangement.SpaceBetween,
            verticalAlignment = Alignment.CenterVertically,
        ) {
            // RootTab.entries 会依次拿到 Home、Favorites、Me。
            RootTab.entries.forEach { tab ->
                Text(
                    text = when (tab) {
                        RootTab.Home -> "首页"
                        RootTab.Favorites -> "收藏"
                        RootTab.Me -> "我的"
                    },
                    color = if (rootTab == tab) Color(0xFF17B59E) else Color(0xFF7D858E),
                    fontWeight = FontWeight.Bold,
                    modifier = Modifier.clickable(enabled = isEnabled) {
                        onRootTab(tab)
                    },
                )
            }
        }
    }
}

0. 开始前先懂目标

最终效果：Mini Player 和 Bottom Tab 不再各自动，而是被一个叫 BottomChrome 的父容器包起来，页面切换时整个父容器一起移动。

你要实现的不是“隐藏某个按钮”，而是建立一套底部区域的规则。这个规则会回答三个问题：

  • 当前页面要不要显示 Bottom Tab？
  • 当前页面要不要显示 Mini Player？
  • 页面内容底部应该留多少空间，才不会被底部控件挡住？
为什么要这样做：如果每个页面自己决定显示隐藏，很快会出现很多零散判断。后面加播放页、设置页、歌词页时，底部动画会越来越难维护。

1. 用人话理解：不要移动两块积木，要移动整个托盘

先把 Mini Player 和 Bottom Tab 想象成放在同一个托盘上的两块积木。 用户切换页面时，应该看到的是“托盘”上下移动，而不是两块积木各自乱跑。 

BottomChrome 这个托盘
├── MiniPlayer      第一块积木：正在播放的小条
└── BottomNavigation 第二块积木：首页 / 收藏 / 我的
首页 托盘完全露出来，所以 Mini Player 和 Bottom Tab 都看得见。
专辑详情页 托盘向下移动一点，Bottom Tab 掉到屏幕外，只剩 Mini Player 露出来。
播放页 / 设置页 托盘继续向下移动，Mini Player 和 Bottom Tab 都离开屏幕。

2. 你要改哪些文件

在当前 KMP Music 项目里，只需要关注这 4 个文件：

文件 它负责什么 你会做什么
MusicAppModels.kt 定义页面、导航、全局 UI 状态 新增 ChromeMode，告诉 App 当前页面底部区域该怎么显示
MusicTheme.kt 统一管理尺寸、颜色、字体 确认 Mini Player 和 Bottom Tab 的高度 token
MusicApp.kt 真正把页面、Mini Player、Bottom Tab 画出来 新增 BottomChrome 父容器，让它整体移动
MusicAppControllerTest.kt 测试导航和播放状态 加测试，确认不同页面对应正确的底部 Chrome 状态
小白检查：如果你不知道文件在哪里，可以在编辑器里搜索文件名。不要同时打开太多文件，按下面步骤一个一个改。

3. 复刻前置条件：先确认你手里有没有这些东西

重要结论：如果是在 KMP Music 当前项目里，下面步骤可以直接照做；如果是在一个全新项目里，必须先准备这一节的最小骨架，否则复制代码会因为缺少类、函数或 import 而报错。
阅读方法：标着“当前项目”的代码用于 KMP Music；标着“全新项目”的代码用于从零复刻。如果你是在新项目里学习，优先复制“全新项目”代码块。

你不需要先理解所有代码，但要知道这些名字必须存在：

名字 它是什么 没有它会怎样
RootTab 首页 / 收藏 / 我的 三个主 Tab BottomNavigation 不知道有几个 tab
SecondaryScreen 搜索、播放页、设置页等二级页面 NavigationState 无法判断当前页面属于哪类
NavigationState 当前导航状态 无法派生 AppChromeMode
Song 当前歌曲数据 MiniPlayer 不知道显示哪首歌
MiniPlayer 底部播放小条 BottomChrome 里没有第一块内容
BottomNavigation 底部主导航 BottomChrome 里没有第二块内容

如果你是全新项目，先复制这个最小导航骨架

放在哪里：可以新建一个 AppModels.kt。如果你的项目已有类似文件，就放到已有导航模型文件里。

enum class RootTab {
    Home,
    Favorites,

把底部显示规则合成一个 AppChromeMode

为什么：一个页面的底部规则不只一个值。它同时包括：Tab 是否显示、BottomChrome 在哪里、内容留白是多少。

复制这段 
/**
 * 页面全局 chrome 策略，统一管理底部 Tab、迷你播放器和页面留白。
 * 一个页面只对应一种 AppChromeMode，后续所有底部 UI 都读它。
 */
enum class AppChromeMode(
    // BottomNavigation 是否可见/可点。
    val showsBottomNavigation: Boolean,
    // BottomChrome 父容器应该停在哪里。
    val bottomChromePlacement: BottomChromePlacement,
    // 页面内容底部应该留多少空间。
    val contentBottomSpace: ContentBottomSpace,
) {
    // 一级页：MiniPlayer + BottomNavigation 都显示。
    TopLevel(
        showsBottomNavigation = true,
        bottomChromePlacement = BottomChromePlacement.TopLevel,
        contentBottomSpace = ContentBottomSpace.TopLevel,
    ),
    // 普通二级页：隐藏 BottomNavigation，只露出 MiniPlayer。
    SecondaryWithMiniPlayer(
        showsBottomNavigation = false,
        bottomChromePlacement = BottomChromePlacement.MiniPlayerOnly,
        contentBottomSpace = ContentBottomSpace.SecondaryWithMiniPlayer,
    ),
    // 沉浸二级页：整个底部 Chrome 都隐藏。
    SecondaryFullscreen(
        showsBottomNavigation = false,
        bottomChromePlacement = BottomChromePlacement.Hidden,
        contentBottomSpace = ContentBottomSpace.Fullscreen,
    ),
}
为什么要合成一个：这样页面内容、Mini Player、Bottom Tab 都读取同一份规则，不会出现“控件隐藏了，但内容还留着空白”的错位。

新增 BottomChrome：真正的“底部托盘”

还是文件：MusicApp.kt

放在哪里：放在 MiniPlayer 函数上方。

为什么：之前 Mini Player 和 Bottom Tab 是兄弟节点，会各自动。现在要把它们变成父子结构，由父容器统一移动。

复制这段 
// BottomChrome 是父容器，负责整体动画。
@Composable
private fun BottomChrome(
    song: Song,
    isPlaying: Boolean,
    placement: BottomChromePlacement,
    showsBottomNavigation: Boolean,
    rootTab: RootTab,
    onOpen: () -> Unit,
    onToggle: () -> Unit,
    onPrev: () -> Unit,
    onQueue: () -> Unit,
    onRootTab: (RootTab) -> Unit,
    modifier: Modifier = Modifier,
) {
    // 唯一的位移动画。不要再给 MiniPlayer 或 BottomNavigation 单独写位移动画。
    val stackOffset: Dp by animateDpAsState(
        targetValue = when (placement) {
            // 一级页不移动。
            BottomChromePlacement.TopLevel -> 0.dp
            // 普通二级页下移一个 Tab 高度。
            BottomChromePlacement.MiniPlayerOnly -> scaledDp(MusicDimens.BottomNavHeight)
            // 沉浸页下移 MiniPlayer + Tab 总高度。
            BottomChromePlacement.Hidden -> scaledDp(MusicDimens.BottomNavHeight + MusicDimens.MiniPlayerHeight)
        },
        animationSpec = tween(durationMillis = 780, easing = FastOutSlowInEasing),
        label = "BottomChromeOffset",
    )
    Box(
        modifier = modifier
            .fillMaxWidth()
            .navigationBarsPadding(),
    ) {
        // 这个 Box 是真正会移动的“底部托盘”。
        Box(
            modifier = Modifier
                .align(Alignment.BottomCenter)
                .fillMaxWidth()
                .height(scaledDp(MusicDimens.MiniPlayerHeight + MusicDimens.BottomNavHeight))
                .offset(y = stackOffset),
        ) {
            // MiniPlayer 固定在托盘顶部。
            MiniPlayer(
                song = song,
                isPlaying = isPlaying,
                onOpen = onOpen,
                onToggle = onToggle,
                onPrev = onPrev,
                onQueue = onQueue,
                modifier = Modifier.align(Alignment.TopCenter),
            )
            // BottomNavigation 固定在托盘底部。
            BottomNavigation(
                rootTab = rootTab,
                isEnabled = showsBottomNavigation,
                onRootTab = onRootTab,
                modifier = Modifier.align(Alignment.BottomCenter),
            )
        }
    }
}
关键理解：动画只在 BottomChrome 上。Mini Player 和 Bottom Tab 自己不再上下移动，所以用户看到的是整体运动。

6. 常见错误：哪里最容易做错？

错误 1：Mini Player 和 Bottom Tab 还在分别 animate

表现：两个控件像两块东西分别滑动。修复：只保留 BottomChromeOffset 一个整体动画。

错误 2：只隐藏控件，没有改内容底部留白

表现：播放页或设置页底部空出一大截。修复：内容 padding 必须读取 chromeMode.contentBottomSpace。

错误 3：在页面里写 if 判断

表现：每个页面都有自己的特殊逻辑。修复：把页面到 ChromeMode 的映射放回 NavigationState。

错误 4：BottomNavigation 隐藏后还能点

表现：普通二级页看不到 Tab，但点击底部区域可能误触。修复：给 BottomNavigation 传 isEnabled = showsBottomNavigation。

7. 交互验收：照着点一遍

操作路径 正确效果 错了说明什么
首页 → 专辑详情 整个底部托盘下移，只露出 Mini Player，Tab 不见。 如果 Mini 和 Tab 分开动，说明父容器没有统一动画。
专辑详情 → 首页 整个底部托盘回来，Mini Player 正好在 Tab 上方，没有缝。 如果有缝，说明高度或 padding 不对。
首页 → 播放页 Mini Player 和 Tab 作为整体离开屏幕。 如果 Mini 直接消失，说明用了局部 if，而不是整体位移。
播放页 → 首页 整个底部托盘回来，不闪、不跳、不重叠。 如果闪动，检查是否还有多个 animate。
首页 → 设置页 不显示 Mini Player，页面内容底部也没有奇怪空白。 如果底部空白大，检查 ContentBottomSpace。

8. 术语表：忘了就回来看

术语 小白解释
MiniPlayer 底部显示当前歌曲的小播放条。
BottomNavigation 首页、收藏、我的这三个主导航。
BottomChrome 装着 Mini Player 和 Bottom Tab 的父容器，也就是“底部托盘”。
ChromeMode 当前页面的底部规则：显示什么、移动到哪里、内容留多少空间。
TopLevel 一级页状态，Mini Player 和 Bottom Tab 都显示。
MiniPlayerOnly 普通二级页状态，只露出 Mini Player。
Hidden 沉浸页面状态，整个底部托盘都藏起来。
最终记住这句话就够了：页面只负责说“我是谁”，NavigationState 负责算出 ChromeMode，BottomChrome 负责整体运动。

这份教程基于 KMP Music 当前实现沉淀，可用于同类音乐 App、视频 App、带全局底部控制条的移动端项目。
